## Nun noch die Vote-Funktion ```typescript|contenteditable // imports/Article.ts export default class Article { ... votes : number; constructor(title, link) { ... this.votes = 0; } voteUp() { this.votes += 1; } voteDown() { this.votes -= 1; } } ```
```typescript|contenteditable // imports/RankingBook.ts import { Component, View, EventEmitter } from 'angular2/angular2'; ... @Component({ ... events: ['voted'] }) @View({ template: ` <book> <div class="votes">{{ book.votes }}</div> <div class="main"> ... <ul> <li><a href (click)="voteUp()">upvote</a></li> <li><a href (click)="voteDown()">downvote</a></li> </ul> </div> </book>` }) ```
```typescript|contenteditable // imports/RankingBook.ts ... export default class RankingBook { ... voted: EventEmitter; constructor() { this.voted = new EventEmitter(); } voteUp(){ this.book.voteUp(); this.voted.next(this.book); return false; // stop event propagation, avoid page reload } voteDown(){ this.book.voteDown(); this.voted.next(this.book); return false; // stop event propagation, avoid page reload } } ```
