## Anpassen unserer BookRankingApp ```typescript/contenteditable // imports/BookRankingApp.ts import { Component, View, NgFor } from 'angular2/angular2'; import Book from 'imports/Book'; import RankingBook from 'imports/RankingBook'; ... ```
```typescript/contenteditable // imports/BookRankingApp.ts ... @View({ directives: [RankingBook, NgFor], template: ` <section class="new-link"> <div class="control-group"> <div><label for="title">Title:</label></div> <div><input name="title" #newtitle></div> </div> <div class="control-group"> <div><label for="link">Link:</label></div> <div><input name="link" #newlink/></div> </div> <button (click)="addBook(newtitle, newlink)">Submit</button> </section> <book *ng-for="#book of books" [book]="book"></book> ` }) ... ```
```typescript/contenteditable // imports/BookRankingApp.ts ... export default class BookRankingApp { books: Array constructor() { this.books = [ new Book('ExampleBook', 'http://mybook.org'), new Book('AnotherExample', 'http://yourbook.com') ] } addBook(title, link) { var newBook = new Book(title.value, link.value); this.books.push(newBook); title.value = ''; link.value = ''; } } ```
## Nun noch die Vote-Funktion ```typescript/contenteditable // imports/Article.ts export default class Article { ... votes : number; constructor(title, link) { ... this.votes = 0; } voteUp() { this.votes += 1; } voteDown() { this.votes -= 1; } } ```
```typescript/contenteditable // imports/RankingBook.ts import { Component, View, EventEmitter } from 'angular2/angular2'; ... @Component({ ... events: ['voted'] }) @View({ template: ` <book> <div class="votes">{{ book.votes }}</div> <div class="main"> ... <ul> <li><a href (click)="voteUp()">upvote</a></li> <li><a href (click)="voteDown()">downvote</a></li> </ul> </div> </book>` }) ```
```typescript/contenteditable // imports/RankingBook.ts ... export default class RankingBook { ... voted: EventEmitter; constructor() { this.voted = new EventEmitter(); } voteUp(){ this.book.voteUp(); this.voted.next(this.book); return false; // stop event propagation, avoid page reload } voteDown(){ this.book.voteDown(); this.voted.next(this.book); return false; // stop event propagation, avoid page reload } } ```
## Und nun noch sortieren ```typescript/contenteditable // imports/BookRankingApp.ts ... @View({ directives: [RankingBook, NgFor], template: ` ... <book (voted)="sortList(article)"></book> ` }) export default class BookRankingApp { ... addBook(title, link) { ... this.sortList(newBook); } sortList(book) { this.books.sort((a, b) => b.votes - a.votes); } } ```
